pub trait CLike {
    fn to_u32(&self) -> u32;
    unsafe fn from_u32(u32) -> Self;
}

An interface for casting C-like enum to u32 and back. A typical implementation can be seen below:

use std::mem;

#[derive(Copy)]
#[repr(u32)]
enum Foo {
    A, B, C
}

impl CLike for Foo {
    fn to_u32(&self) -> u32 {
        *self as u32
    }
    unsafe fn from_u32(v: u32) -> Foo {
        mem::transmute(v)
    }
}
