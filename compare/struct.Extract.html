A comparator that extracts a sort key from a value.

Examples

use compare::{Compare, Extract, natural};
use std::cmp::Ordering::Greater;

let a = vec![1, 2, 3];
let b = vec![4, 5];

let cmp = Extract::new(|vec: &Vec<u8>| vec.len(), natural());
assert_eq!(cmp.compare(&a, &b), Greater);
